00080 // The COMPILE_ASSERT macro can be used to verify that a compile time
00081 // expression is true. For example, you could use it to verify the
00082 // size of a static array:
00083 //
00084 //   COMPILE_ASSERT(ARRAYSIZE(content_type_names) == CONTENT_NUM_TYPES,
00085 //                  content_type_names_incorrect_size);
00086 //
00087 // or to make sure a struct is smaller than a certain size:
00088 //
00089 //   COMPILE_ASSERT(sizeof(foo) < 128, foo_too_large);
00090 //
00091 // The second argument to the macro is the name of the variable. If
00092 // the expression is false, most compilers will issue a warning/error
00093 // containing the name of the variable.
00094 
00095 template <bool>
00096 struct StubsCompileAssert {
00097 };
