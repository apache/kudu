288 inline int Slice::compare(const Slice& b) const {
289  const int min_len = (size_ < b.size_) ? size_ : b.size_;
290  int r = MemCompare(data_, b.data_, min_len);
291  if (r == 0) {
292  if (size_ < b.size_) r = -1;
293  else if (size_ > b.size_) r = +1;
294  }
295  return r;
296 }
